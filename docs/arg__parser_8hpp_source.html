20 inline std::pair<std::filesystem::path, std::vector<std::string>> setup_and_get_posargs(int argc, char** argv) {
21  namespace po = boost::program_options;
22  namespace fs = std::filesystem;
23  std::vector<std::string> posargs{};
24  fs::path outdir{};
25 
26  const auto filename = fs::path(argv[0]).filename().string();
27  po::options_description desc{"magnum.af ArgParser\nUsage: " + filename + " [options]"};
28  desc.add_options()("help,h", "Produce help message.");
29  desc.add_options()("outdir,o", po::value<fs::path>(&outdir)->default_value("output_" + filename),
30  "Output directory. Will be created and is accessible via ArgParser.outdir(). Defaults to "
31  "'output_<binaryname>'.");
32  desc.add_options()("no-overwrite,n", "Abort if outdir already exists. Prevents file overwriting.");
33  desc.add_options()("backend,b", po::value<std::string>(),
34  "'cuda', 'opencl' or 'cpu'. Select arrayfire backend via 'af::setBackend(b)'.");
35  desc.add_options()("device,d", po::value<unsigned>(), "Set 'af::setDevice(d)', e.g. used for selecting a GPU.");
36  desc.add_options()("verbose,v", "Make this parser verbose, printing parsing steps.");
37  desc.add_options()("posargs", po::value<std::vector<std::string>>(&posargs),
38  "Positional arguments, access via 'ArgParser.posargs()'.");
39 
40  po::positional_options_description p;
41  p.add("posargs", -1);
42 
43  try {
44  po::variables_map vm;
45  po::store(po::command_line_parser(argc, argv).options(desc).positional(p).run(), vm);
46  po::notify(vm);
47 
48  if (vm.count("help")) {
49  std::cout << desc << "\n";
50  std::exit(0);
51  }
52 
53  if (vm.count("backend")) {
54  const auto backend = vm["backend"].as<std::string>();
55  if (vm.count("verbose")) {
56  std::cout << "--backend: Setting af::setBackend() to " << backend << ".\n";
57  }
58 
59  if (backend == "cuda" or backend == "opencl" or backend == "cpu") {
60  try {
61  if (backend == "cuda") {
62  af::setBackend(AF_BACKEND_CUDA);
63  }
64  if (backend == "opencl") {
65  af::setBackend(AF_BACKEND_OPENCL);
66  }
67  if (backend == "cpu") {
68  af::setBackend(AF_BACKEND_CPU);
69  }
70 
71  } catch (const std::exception& e) {
72  std::cerr << "--backend: Error while calling af::setBackend() with '" << backend << "'.\n"
73  << "Maybe your hardware does not support the selected backend or drivers are not "
74  "installed.\n";
75  throw;
76  }
77  } else {
78  throw po::validation_error(po::validation_error::invalid_option_value, "backend");
79  }
80  }
81 
82  if (vm.count("device")) {
83  if (vm.count("verbose")) {
84  std::cout << "--device: Setting af::setDevice(" << vm["device"].as<unsigned>() << ").\n";
85  }
86  try {
87  af::setDevice(vm["device"].as<unsigned>());
88  } catch (const std::exception& e) {
89  std::cerr << "--device: Error while calling af::setDevice(" << vm["device"].as<unsigned>()
90  << "). Device number not valid.\n";
91  throw;
92  }
93  }
94 
95  // Location dependent, don't move after --outdir option.
96  if (vm.count("no-overwrite")) {
97  if (vm.count("verbose")) {
98  std::cout << "--no-overwrite: Checking if --outdir " << fs::absolute(outdir) << " exists.\n";
99  }
100  if (fs::exists(fs::path(outdir))) {
101  throw std::runtime_error(
102  "--no-overwrite: Error, active flag '--no-overwrite' prevents writing into already "
103  "existing outdir" +
104  fs::absolute(outdir).string());
105  }
106  }
107 
108  if (vm.count("outdir")) {
109  bool mkdirs = fs::create_directories(outdir);
110  if (vm.count("verbose") and mkdirs) {
111  std::cout << "--outdir: Created outdir " << fs::absolute(outdir) << '\n';
112  } else if (vm.count("verbose")) {
113  std::cout << "--outdir: Outputdir " << fs::absolute(outdir)
114  << " already exists, files could be overwritten.\n";
115  }
116  }
117 
118  if (vm.count("posargs")) {
119  if (vm.count("verbose")) {
120  std::cout << "--posargs: Positional arguments are: " << vm["posargs"].as<std::vector<std::string>>()
121  << '\n';
122  }
123  } else {
124  if (vm.count("verbose")) {
125  std::cout << "--posargs: No positional arguments, ArgParser.posargs() is empty.\n";
126  }
127  }
128 
129  // Writing logging info
130  const auto logfile = outdir / "log.txt";
131  if (vm.count("verbose")) {
132  std::cout << "Logging posargs info for script reproducibility to" << fs::absolute(logfile) << '\n';
133  }
134  std::ofstream ofst(logfile);
135  for (const auto& elem : posargs) {
136  const auto i = &elem - &posargs[0];
137  ofst << "posargs[" << i << "] = " << elem << '\n';
138  }
139  ofst << "GIT_HEAD_SHA1=" << GIT_HEAD_SHA1 << '\n';
140 
141  if (vm.count("verbose")) {
142  af::info();
143  std::cout << "Argument parsing finished.\n";
144  }
145 
146  } catch (const std::exception& e) {
147  std::cerr << "Exception occurred while parsing command line arguments:\n"
148  << e.what() << '\n'
149  << "Showing help (" << filename << " -h):\n"
150  << desc << '\n'
151  << "Aborting...\n";
152  std::exit(1);
153  }
154 
155  return {outdir, posargs};
156 }
