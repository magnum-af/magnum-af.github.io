30  if (a.dims() == b.dims()) {

76  af::array operator/(const af::array& b) const {
77  const auto visitor = overloaded{
78  [&b](double d) { return d / b; },
79  [&b](const af::array& a) {
80  const auto divide = [&b](const af::array& a) {
81  if (a.dims() == b.dims()) {
82  return a / b;
83  } else if (a.dims(0) == b.dims(0) and a.dims(1) == b.dims(1) and a.dims(2) == b.dims(2) and
84  b.dims(3) == 3) {
85  return af::tile(a, 1, 1, 1, 3) / b;
86  } else {
87  throw std::runtime_error("DoubleOrArray::operator/: array dims do not match.");
88  }
89  };
90  auto result = divide(a);
91  af::replace(result, b != 0, 0); // replacing potential divs by null (i.e. NaN) with zeros.
92  // Optional optimization: cache whether a has zero vals
93  return result;
94  },
95  };

101  af::array operator()(af::dim4 dims, af::dtype type) const {
102  const auto visitor = overloaded{
103  [&](double d) { return af::constant(d, dims, type); },
104  [&](const af::array& a) {
105  if (a.dims() == dims) {
106  return a.as(type);
107  } else if (a.dims(0) == dims[0] and a.dims(1) == dims[1] and a.dims(2) == dims[2] and dims[3] == 3) {
108  return af::tile(a, 1, 1, 1, 3).as(type);
109  } else {
110  throw std::runtime_error("DoubleOrArray::operator(): array dims do not match.");
111  }
112  },
113  };
