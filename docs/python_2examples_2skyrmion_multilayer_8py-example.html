46 def setup_init_skyrmion(dims_3d : [int], R = 0.25):
47  """ set m [0,0,-1] for r <= R, else [0,0,1]"""
48  nx, ny, nz = dims_3d[0], dims_3d[1], dims_3d[2]
49  m = np.zeros((nx, ny, nz, 3))
50  for ix in range (0, nx):
51  for iy in range(0, ny):
52  a= nx/2
53  b= ny/2
54  rx=ix-nx/2.
55  ry=iy-ny/2.
56  r = pow(rx, 2)/pow(a, 2)+pow(ry, 2)/pow(b, 2)
57  if(r <= R):
58  m[ix, iy, :, 2] = -1.
59  else:
60  m[ix, iy, :, 2] = 1.
61  return af.from_ndarray(m)
